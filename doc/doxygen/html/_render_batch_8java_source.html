81 public void start() {
82 // Generate and bind a Vertex Array Object
83 vaoID = glGenVertexArrays();

131 public void render() {
132 boolean rebufferData = false;
133 for (int i = 0; i < numSprites; i++) {
134 SpriteRenderer spr = sprites[i];
135 if (spr.isDirty()) {
136 loadVertexProperties(i);
137 spr.setClean();
138 rebufferData = true;
139 }
140 }
141
142 if (rebufferData) {
143 glBindBuffer(GL_ARRAY_BUFFER, vboID);
144 glBufferSubData(GL_ARRAY_BUFFER, 0, vertices);
145 }
146
147 // Use shader
148
149 shader.use();
150 shader.uploadMat4f("uProjection", Window.getScene().camera().getProjectionMatrix());
151 shader.uploadMat4f("uView", Window.getScene().camera().getViewMatrix());
152
153 for (int i = 0; i < textures.size(); i++) {
154 // First slot (0) reserved for no Texture
155 glActiveTexture(GL_TEXTURE0 + i + 1);
156 textures.get(i).bind();
157 }
158
159 shader.uploadIntArray("uTextures", texSlots);
160
161 glBindVertexArray(vaoID);
162
163 glEnableVertexAttribArray(0);
164 glEnableVertexAttribArray(1);
165
166 glDrawElements(GL_TRIANGLES, this.numSprites * 6, GL_UNSIGNED_INT, 0);
167
168 glDisableVertexAttribArray(0);
169 glDisableVertexAttribArray(1);
170 glBindVertexArray(0);
171
172 for (int i = 0; i < textures.size(); i++) {
173 textures.get(i).unbind();
174 }
175
176 shader.detach();
177 }

179 private void loadVertexProperties(int index) {
180 SpriteRenderer sprite = this.sprites[index];
181
182 // Find the offset within array (4 vertivces per sprite)
183 int offset = index * 4 * VERTEX_SIZE;
184 // Float float float, flaot, float, float
185
186 Vector4f color = sprite.getColor();
187 Vector2f[] texCoords = sprite.getTexCoords();
188
189 int texID = 0;
190 if (sprite.getTexture() != null) {
191 for (int i = 0; i < textures.size(); i++) {
192 if (textures.get(i) == sprite.getTexture()) {
193 texID = i + 1;
194 break;
195 }
196 }
197 }
198
199 // Add vertice with appropriate properties
200
201 float xAdd = 1.0f;
202 float yAdd = 1.0f;
203
204 for (int i = 0; i < 4; i++) {
205 if (i == 1) {
206 yAdd = 0.0f;
207 } else if (i == 2) {
208 xAdd = 0.0f;
209 } else if (i == 3) {
210 yAdd = 1.0f;
211 }
212
213 // Load Position
214 vertices[offset] = sprite.gameObject.transform.position.x + (xAdd * sprite.gameObject.transform.scale.x);
215 vertices[offset + 1] = sprite.gameObject.transform.position.y
216 + (yAdd * sprite.gameObject.transform.scale.y);
217
218 // Load COlor
219 vertices[offset + 2] = color.x; // R
220 vertices[offset + 3] = color.y; // B
221 vertices[offset + 4] = color.z; // G
222 vertices[offset + 5] = color.w; // A
223
224 // Load position
225 vertices[offset + 6] = texCoords[i].x; // G
226 vertices[offset + 7] = texCoords[i].y; // A
227
228 // Load texture id
229 vertices[offset + 8] = texID;
230
231 offset += VERTEX_SIZE;
232 }
233 }
