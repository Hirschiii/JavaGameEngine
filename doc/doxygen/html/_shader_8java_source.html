49 public Shader(String filepath) {
50 this.filepath = filepath;
51 try {
52 String source = new String(Files.readAllBytes(Paths.get(filepath)));
53 String[] splitString = source.split("(#type)( )+([a-zA-Z]+)");
54
55 // FInd the first pattern after #type
56 int index = source.indexOf("#type") + 6;
57 int eol = source.indexOf("\n", index);
58 String firstPattern = source.substring(index, eol).trim();
59
60 // FInd the second pattern after #type
61 index = source.indexOf("#type", eol) + 6;
62 eol = source.indexOf("\n", index);
63
64 String secondPattern = source.substring(index, eol).trim();
65
66 if (firstPattern.equals("vertex")) {
67 vertexSource = splitString[1];
68 } else if (firstPattern.equals("fragment")) {
69 fragmentSource = splitString[1];
70 } else {
71 throw new IOException("Unexpected Token '" + firstPattern + "'");
72 }
73
74 if (secondPattern.equals("vertex")) {
75 vertexSource = splitString[2];
76 } else if (secondPattern.equals("fragment")) {
77 fragmentSource = splitString[2];
78 } else {
79 throw new IOException("Unexpected Token '" + secondPattern + "'");
80 }
81
82 } catch (IOException e) {
83 e.printStackTrace();
84 assert false : "Error : Could not open file for Shader : '" + filepath + "'";
85 }
86
87 }
